void Tao_Mang()
{
	// Tao mang co sl pt >=15 
	srand(time(NULL));
	int n,a[100];
	n = 15 + rand() % (20 - 15 + 1);
	for (int i = 0; i < n; i++)
	{
		a[i] = rand() % (15 + 1);
	}
	for (int i = 0; i < n; i++)
	{
		cout << a[i] << " ";
	}
	cout << endl;
}

void Linear_Search()
{
	// linear seach 

	int n, tmp, dem = 0, a[100];
	cout << "Vui long nhap so luong phan tu " << endl;
	cin >> n;
	cout << " Vui long nhap mang " << endl;

	for (int i = 0; i < n; i++)
	{
		cin >> a[i];
	}
	cout << "Vui long nhap so can tim " << endl;
	cin >> tmp;
	for (int i = 0; i < n; i++)
	{
		if (tmp == a[i])
		{
			dem++;
		}
	}
	if (dem > 0)
	{
		cout << "Da tim thay " << endl;
	}
	else
	{
		cout << "Khong tim thay" << endl;
	}
}
